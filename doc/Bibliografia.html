def get_autores
   s = ""
   i=0

   while i < @autores.size
        nombre, apellido = @autores[i].split(/\W+/)
        nombre = nombre[0] + "."
        autor = apellido + ', ' + nombre
        s += autor

        if !(i == @autores.size-1)
            s += " & "
        end
        i += 1
   end

   return s
end
